APPROACH 1 — Two Pointers from ends
function moveZeroes(arr) { let left = 0; let right = arr.length - 1; while (left < right) { if (arr[right] !== 0) { [arr[left], arr[right]] = [arr[right], arr[left]]; left++; } else { right--; } } return arr; }
