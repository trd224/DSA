Why it works: If a majority element exists (appears > n/2 times), even if every other element tries to cancel it out 1-for-1, the majority element still has votes left over. The candidate surviving at the end must be the majority.
majorityElement.js 
function majorityElement(nums) {
  let count     = 0;
  let candidate = null;
 
  for (let num of nums) {
 
    if (count === 0) {
      candidate = num;   // pick new candidate
    }
 
    if (num === candidate) {
      count++;             // vote FOR candidate
    } else {
      count--;             // vote AGAINST candidate
    }
  }
 
  return candidate;       // survivor = majority
}
