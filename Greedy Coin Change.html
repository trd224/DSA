function takeCoin(arr, total) {
  const coinsArr = [];
  for (let i = 0; i < arr.length; i++) {
    while (total >= arr[i]) {
      coinsArr.push(arr[i]);  // take this coin
      total = total - arr[i]; // subtract
    }                          // repeat while fits
  }                            // next denomination
  return coinsArr;
}
