function missingNumber(nums) {
  let xor = 0;
  let n   = nums.length;   // n = ${n}
 
  // Pass 1: XOR all indices 0..n
  for (let i = 0; i <= n; i++) {
    xor ^= i;   // XOR with each index
  }
 
  // Pass 2: XOR all array values
  for (let num of nums) {
    xor ^= num; // cancels every num that exists
  }
 
  return xor;   // only missing number remains!
}
