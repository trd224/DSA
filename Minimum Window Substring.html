function minWindow(s, t) {
  let map = {};                              // freq map of t
  for (let char of t)
    map[char] = (map[char] || 0) + 1;
 
  let left = 0, count = 0;
  let minLen = Infinity, start = 0;
 
  for (let right = 0; right < s.length; right++) {
    if (map[s[right]] !== undefined) {      // char is in t
      map[s[right]]--;                     // decrement count
      if (map[s[right]] >= 0) count++;    // still needed → match!
    }
    while (count === t.length) {           // valid window!
      if (right - left + 1 < minLen) {    // new best?
        minLen = right - left + 1;
        start = left;
      }
      if (map[s[left]] !== undefined) {    // left char is in t
        map[s[left]]++;                   // restore count
        if (map[s[left]] > 0) count--;   // was needed → unmatch
      }
      left++;                              // shrink window
    }
  }
  return minLen === Infinity
    ? ""
    : s.substring(start, start + minLen);
}
