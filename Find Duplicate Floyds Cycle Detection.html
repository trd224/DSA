function findDuplicate(nums) {
  // Phase 1: detect cycle — slow=1 step, fast=2 steps
  let slow = nums[0];
  let fast = nums[0];
 
  do {
    slow = nums[slow];            // 1 step
    fast = nums[nums[fast]];      // 2 steps
  } while (slow !== fast);
 
  // Phase 2: find entrance — reset fast to start
  fast = nums[0];
 
  while (slow !== fast) {
    slow = nums[slow];            // both move 1 step
    fast = nums[fast];
  }
 
  return slow;  // = duplicate number
}
